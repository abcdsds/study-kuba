apiVersion: apps/v1
kind: Deployment
metadata:
  name: health-check-app
  namespace: kimmunki
  labels:
    app: health-check-app
spec:
  replicas: 2
  selector:
    matchLabels:
      app: health-check-app
  template:
    metadata:
      labels:
        app: health-check-app
    spec:
      containers:
      - name: health-check-app
        image: asia.gcr.io/kuba-310814/kimmunki-app:0.1
        ports:
        - containerPort: 8080

---

apiVersion: v1
kind: Service
metadata:
  name: health-check-app-service
  namespace: kimmunki
spec:
  selector:
    app: health-check-app
  ports:
    - protocol: TCP
      port: 8080
      targetPort: 8080

---

kubectl port-forward svc/health-check-app-service 8080
